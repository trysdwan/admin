// ==================== MAIN DO POST HANDLER ====================
function doPost(e) {
  try {
    const action = e.parameter.action;
    
    if (action === 'addPost') {
      return handleAddPost(e.parameter);
    } else if (action === 'updatePost') {
      return handleUpdatePost(e.parameter);
    } else if (action === 'deletePost') {
      return handleDeletePost(e.parameter);
    } else if (action === 'like') {
      return handleLike(e.parameter.postId);
    } else if (action === 'comment') {
      return handleComment(e.parameter.postId, e.parameter.userName, e.parameter.commentText);
    } else if (action === 'registerAuthor') {
      return handleRegisterAuthor(e.parameter);
    } else if (action === 'requestPasswordReset') {
      return handleRequestPasswordReset(e.parameter);
    } else if (action === 'verifyResetOtp') {
      return handleVerifyResetOtp(e.parameter);
    } else if (action === 'resetPassword') {
      return handleResetPassword(e.parameter);
    } else if (action === 'updateAuthorStatus') {
      return handleUpdateAuthorStatus(e.parameter);
    } else if (action === 'updateAuthorProfile') {
      return handleUpdateAuthorProfile(e.parameter);
    }
    
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: 'Invalid action: ' + action }))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch(error) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: error.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return doPost(e);
}

// ==================== POST MANAGEMENT FUNCTIONS ====================

// Handle adding new post
function handleAddPost(params) {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(BLOG_SHEET);
  
  const requiredFields = ['postId', 'category', 'author', 'title', 'content'];
  for (let field of requiredFields) {
    if (!params[field]) {
      return ContentService
        .createTextOutput(JSON.stringify({ success: false, error: `Missing field: ${field}` }))
        .setMimeType(ContentService.MimeType.JSON);
    }
  }
  
  // Check if post ID already exists
  const existingData = sheet.getDataRange().getValues();
  for (let i = 1; i < existingData.length; i++) {
    if (String(existingData[i][0]) === String(params.postId)) {
      return ContentService
        .createTextOutput(JSON.stringify({ success: false, error: 'Post ID already exists' }))
        .setMimeType(ContentService.MimeType.JSON);
    }
  }
  
  const newRow = [
    params.postId, 
    params.category, 
    params.tags || '', 
    params.author,
    params.publishedTime || new Date().toLocaleString(), 
    params.title, 
    params.content,
    parseInt(params.likeCount) || 0, 
    parseInt(params.shareCount) || 0
  ];
  
  sheet.appendRow(newRow);
  
  return ContentService
    .createTextOutput(JSON.stringify({ success: true, message: 'Post created', postId: params.postId }))
    .setMimeType(ContentService.MimeType.JSON);
}

// Handle updating a post
function handleUpdatePost(params) {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(BLOG_SHEET);
  const data = sheet.getDataRange().getValues();
  let foundRow = -1;
  
  // Find the row with matching post ID
  for(let i = 1; i < data.length; i++) {
    if(String(data[i][0]) === String(params.postId)) {
      foundRow = i + 1; // +1 because sheets are 1-indexed
      break;
    }
  }
  
  if(foundRow === -1) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: 'Post not found' }))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  // Update each column
  sheet.getRange(foundRow, 2).setValue(params.category);  // Column B: Category
  sheet.getRange(foundRow, 3).setValue(params.tags || ''); // Column C: Tags
  sheet.getRange(foundRow, 4).setValue(params.author);     // Column D: Author
  sheet.getRange(foundRow, 6).setValue(params.title);      // Column F: Title
  sheet.getRange(foundRow, 7).setValue(params.content);    // Column G: Content
  
  return ContentService
    .createTextOutput(JSON.stringify({ success: true, message: 'Post updated' }))
    .setMimeType(ContentService.MimeType.JSON);
}

// Handle deleting a post and its comments
function handleDeletePost(params) {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(BLOG_SHEET);
  const commentsSheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(COMMENTS_SHEET);
  const data = sheet.getDataRange().getValues();
  let foundRow = -1;
  
  for(let i = 1; i < data.length; i++) {
    if(String(data[i][0]) === String(params.postId)) {
      foundRow = i + 1;
      break;
    }
  }
  
  if(foundRow === -1) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: 'Post not found' }))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  // Delete the post row
  sheet.deleteRow(foundRow);
  
  // Delete all comments for this post
  const commentsData = commentsSheet.getDataRange().getValues();
  for(let i = commentsData.length - 1; i >= 1; i--) {
    if(String(commentsData[i][0]) === String(params.postId)) {
      commentsSheet.deleteRow(i + 1);
    }
  }
  
  return ContentService
    .createTextOutput(JSON.stringify({ success: true, message: 'Post and comments deleted' }))
    .setMimeType(ContentService.MimeType.JSON);
}

// Handle like increment
function handleLike(postId) {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(BLOG_SHEET);
  const data = sheet.getDataRange().getValues();
  let foundRow = -1;
  
  for(let i = 1; i < data.length; i++) {
    if(String(data[i][0]) === String(postId)) {
      foundRow = i + 1;
      break;
    }
  }
  
  if(foundRow === -1) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: 'Post not found' }))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  const currentLikes = parseInt(data[foundRow - 1][7]) || 0;

  return ContentService
    .createTextOutput(JSON.stringify({ success: true, newLikes: newLikes }))
    .setMimeType(ContentService.MimeType.JSON);
}

// ==================== REGISTRATION FUNCTION ====================

function handleRegisterAuthor(params) {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(PROFILES_SHEET);
  
  // Validate required fields
  const requiredFields = ['authorName', 'email', 'password'];
  for (let field of requiredFields) {
    if (!params[field]) {
      return ContentService
        .createTextOutput(JSON.stringify({ success: false, error: `Missing field: ${field}` }))
        .setMimeType(ContentService.MimeType.JSON);
    }
  }
  
  // Check if author name or email already exists
  const existingData = sheet.getDataRange().getValues();
  for (let i = 1; i < existingData.length; i++) {
    if (existingData[i][0] && existingData[i][0].trim().toLowerCase() === params.authorName.toLowerCase()) {
      return ContentService
        .createTextOutput(JSON.stringify({ success: false, error: 'Author name already exists' }))
        .setMimeType(ContentService.MimeType.JSON);
    }
    if (existingData[i][6] && existingData[i][6].trim().toLowerCase() === params.email.toLowerCase()) {
      return ContentService
        .createTextOutput(JSON.stringify({ success: false, error: 'Email already registered' }))
        .setMimeType(ContentService.MimeType.JSON);
    }
  }
  
  // Add new row to Profiles sheet (Columns A to I)
  // A: Author Name, B: Designation, C: About, D: Image URL, E: Password, F: Confirm, G: Mail ID, H: OTP, I: OTP Expiry
  const newRow = [
    params.authorName,           // A: Author Name
    params.designation || '',    // B: Designation
    params.about || '',          // C: About
    params.imageUrl || '',       // D: Image URL
    params.password,             // E: Password
    'No',                        // F: Confirm (Default No - needs admin approval)
    params.email,                // G: Mail ID
    '',                          // H: OTP (empty for now)
    ''                           // I: OTP Expiry (empty for now)
  ];
  
  sheet.appendRow(newRow);
  
  return ContentService
    .createTextOutput(JSON.stringify({ 
      success: true, 
      message: 'Registration successful. Waiting for admin approval.' 
    }))
    .setMimeType(ContentService.MimeType.JSON);
}

// ==================== PASSWORD RESET FUNCTIONS ====================

// Generate random 6-digit OTP
function generateOTP() {
  return Math.floor(100000 + Math.random() * 900000).toString();
}

// Store OTP in sheet with expiry (10 minutes)
function storeOTP(email, otp) {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(PROFILES_SHEET);
  const data = sheet.getDataRange().getValues();
  
  // Find the user by email (column G)
  for (let i = 1; i < data.length; i++) {
    if (data[i][6] && data[i][6].trim().toLowerCase() === email.toLowerCase()) {
      // Store OTP in column H (index 7)
      sheet.getRange(i + 1, 8).setValue(otp);
      // Store expiry timestamp in column I (index 8) - current time + 10 minutes
      const expiry = new Date().getTime() + (10 * 60 * 1000);
      sheet.getRange(i + 1, 9).setValue(expiry);
      return true;
    }
  }
  return false;
}

// Verify OTP
function verifyOTP(email, otp) {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(PROFILES_SHEET);
  const data = sheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][6] && data[i][6].trim().toLowerCase() === email.toLowerCase()) {
      const storedOTP = data[i][7];
      const expiry = data[i][8];
      const currentTime = new Date().getTime();
      
      if (storedOTP && storedOTP.toString() === otp.toString() && expiry && currentTime < parseInt(expiry)) {
        return true;
      }
    }
  }
  return false;
}

// Clear OTP after verification
function clearOTP(email) {
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(PROFILES_SHEET);
  const data = sheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][6] && data[i][6].trim().toLowerCase() === email.toLowerCase()) {
      sheet.getRange(i + 1, 8).setValue('');
      sheet.getRange(i + 1, 9).setValue('');
      return true;
    }
  }
  return false;
}

// Handle password reset request
function handleRequestPasswordReset(params) {
  const email = params.email;
  const username = params.username || '';
  
  if (!email) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: 'Email is required' }))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  // Check if email exists in Profiles sheet
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(PROFILES_SHEET);
  const data = sheet.getDataRange().getValues();
  let userExists = false;
  let userName = '';
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][6] && data[i][6].trim().toLowerCase() === email.toLowerCase()) {
      userExists = true;
      userName = data[i][0];
      break;
    }
  }
  
  if (!userExists) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: 'Email not found' }))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  // Generate and store OTP
  const otp = generateOTP();
  storeOTP(email, otp);
  
  // Send email with OTP
  try {
    const subject = 'Password Reset Request - Blog Studio';
    const body = `Dear ${userName},\n\nYou requested to reset your password for your Blog Studio account.\n\nYour OTP for password reset is: ${otp}\n\nThis OTP is valid for 10 minutes.\n\nIf you did not request this, please ignore this email.\n\nRegards,\nBlog Studio Team`;
    
    MailApp.sendEmail({
      to: email,
      subject: subject,
      body: body
    });
    
    return ContentService
      .createTextOutput(JSON.stringify({ success: true, message: 'OTP sent to your email' }))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch(error) {
    console.error('Email send error:', error);
    // Still return success with OTP for testing purposes
    return ContentService
      .createTextOutput(JSON.stringify({ success: true, message: 'OTP generated', otp: otp }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// Handle OTP verification
function handleVerifyResetOtp(params) {
  const email = params.email;
  const otp = params.otp;
  
  if (!email || !otp) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: 'Email and OTP are required' }))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  if (verifyOTP(email, otp)) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: true, message: 'OTP verified' }))
      .setMimeType(ContentService.MimeType.JSON);
  } else {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: 'Invalid or expired OTP' }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// Handle password reset
function handleResetPassword(params) {
  const email = params.email;
  const newPassword = params.newPassword;
  
  if (!email || !newPassword) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: 'Email and password are required' }))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  if (newPassword.length < 6) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: 'Password must be at least 6 characters' }))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  // Update the password in Profiles sheet
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(PROFILES_SHEET);
  const data = sheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][6] && data[i][6].trim().toLowerCase() === email.toLowerCase()) {
      // Update password in column E (index 4)
      sheet.getRange(i + 1, 5).setValue(newPassword);
      // Clear OTP and expiry
      sheet.getRange(i + 1, 8).setValue('');
      sheet.getRange(i + 1, 9).setValue('');
      
      return ContentService
        .createTextOutput(JSON.stringify({ success: true, message: 'Password reset successful' }))
        .setMimeType(ContentService.MimeType.JSON);
    }
  }
  
  return ContentService
    .createTextOutput(JSON.stringify({ success: false, error: 'User not found' }))
    .setMimeType(ContentService.MimeType.JSON);
}

// Handle update author profile
function handleUpdateAuthorProfile(params) {
  const oldName = params.oldName;
  const newName = params.authorName;
  const designation = params.designation || '';
  const about = params.about || '';
  const imageUrl = params.imageUrl || '';
  const email = params.email;
  
  if (!oldName || !newName || !email) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, error: 'Missing required fields' }))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(PROFILES_SHEET);
  const data = sheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][0] && data[i][0].trim().toLowerCase() === oldName.toLowerCase()) {
      // Update author name (Column A)
      sheet.getRange(i + 1, 1).setValue(newName);
      // Update designation (Column B)
      sheet.getRange(i + 1, 2).setValue(designation);
      // Update about (Column C)
      sheet.getRange(i + 1, 3).setValue(about);
      // Update image URL (Column D)
      sheet.getRange(i + 1, 4).setValue(imageUrl);
      // Update email (Column G)
      sheet.getRange(i + 1, 7).setValue(email);
      
      // Also update all posts with the new author name
      const blogSheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(BLOG_SHEET);
      const blogData = blogSheet.getDataRange().getValues();
      for (let j = 1; j < blogData.length; j++) {
        if (blogData[j][3] && blogData[j][3].trim().toLowerCase() === oldName.toLowerCase()) {
          blogSheet.getRange(j + 1, 4).setValue(newName);
        }
      }
      
      // Also update comments if needed (optional - comments store user name)
      const commentsSheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(COMMENTS_SHEET);
      const commentsData = commentsSheet.getDataRange().getValues();
      for (let j = 1; j < commentsData.length; j++) {
        if (commentsData[j][1] && commentsData[j][1].trim().toLowerCase() === oldName.toLowerCase()) {
          commentsSheet.getRange(j + 1, 2).setValue(newName);
        }
      }
      
      return ContentService
        .createTextOutput(JSON.stringify({ success: true, message: 'Profile updated successfully' }))
        .setMimeType(ContentService.MimeType.JSON);
    }
  }
  
  return ContentService
    .createTextOutput(JSON.stringify({ success: false, error: 'User not found' }))
    .setMimeType(ContentService.MimeType.JSON);
}
